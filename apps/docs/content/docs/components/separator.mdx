## Usage

```tsx
import { Separator } from "@/components/ui/separator";

export function SeparatorDemo() {
  return (
    <View>
      <Text>Above</Text>
      <Separator />
      <Text>Below</Text>
    </View>
  );
}
```

## Props

| Prop          | Type                         | Default        | Description                      |
| ------------- | ---------------------------- | -------------- | -------------------------------- |
| `orientation` | `'horizontal' \| 'vertical'` | `'horizontal'` | The orientation of the separator |
